@app.post("/accounts")
def create_account(request):
    v = CreateAccountView()
    p = CreateAccountPresenter()
    uc = CreateAccountUseCase(db, p)
    c = CreateAccountController(uc)

    c.execute(request)

    return v.response(p.view_model)
